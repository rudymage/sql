CREATE TABLE SELLER(SNUM NUMBER(4),SNAME VARCHAR2(10) NOT NULL,CITY VARCHAR2(10) NOT NULL,COMM NUMBER(7,2) NOT NULL,PRIMARY KEY(SNUM));
CREATE TABLE CUSTOMER(CNUM NUMBER(4),CNAME VARCHAR2(10) NOT NULL,CITY VARCHAR2(10) NOT NULL,RATING NUMBER(3) NOT NULL,PRIMARY KEY(CNUM));
CREATE TABLE ORDERS(ONUM NUMBER(4),AMT VARCHAR2(10) NOT NULL,ODATE DATE NOT NULL,CNUM NUMBER(4) NOT NULL,SNUM NUMBER(4) NOT NULL,PRIMARY KEY(ONUM));
